TAREA #11: CONVERSION DE TEMPERATURA

Elabore un programa en JAVA que mediante un Menú de opciones permita calcular la conversión de escala de temperatura:
1. De grados Centígrados a grados Farenheit
2. De grados Farenheita grados Centígrados
(Nota: Usar el cuadro de dialogo, showOptionDialog(), para seleccionar : °C -> °F y otro °F -> °C, ) 

            import javax.swing.JOptionPane;

public class tarea11 {
    public static void main(String[] args) {
        int menu;
        menu = JOptionPane.showOptionDialog(null, "Escoge una opcion", "Conversor de Temperatura",
                JOptionPane.DEFAULT_OPTION, JOptionPane.PLAIN_MESSAGE, null, new Object[] { "C -> F", "F -> C" },
                "C a F");

        switch (menu) {
            case 1:
                String entradaCelsius = JOptionPane.showInputDialog("Ingresa la cantidad de Centigrados:");
                int centigrados = Integer.parseInt(entradaCelsius);
                float resultadoFahrenheit = (centigrados * 9 / 5) + 32;
                JOptionPane.showMessageDialog(null, "Conversion de Centigrados a Fahrenheit: " + resultadoFahrenheit);
                break;

            case 2:
                String entradaFahrenheit = JOptionPane.showInputDialog("Ingresa la cantidad de grados Fahrenheit:");
                int fahrenheit = Integer.parseInt(entradaFahrenheit);
                float resultadoCelsius = (fahrenheit - 32) * 5 / 9;
                JOptionPane.showMessageDialog(null, "Conversion de Fahrenheit a Centigrados: " + resultadoCelsius);
                break;

            default:
                JOptionPane.showMessageDialog(null, "Ingresa un numero valido");
                break;
        }
    }
}
