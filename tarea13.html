TAREA #13: SERIE A

Construya un programa en JAVA. Que calcule e imprima la siguiente serie para los N primeros t√©rminos: 1, 8, 243, 16384, 1953125 

            import java.util.Scanner;

public class tarea13 {
    public static void main(String[] args) {
        Scanner lec = new Scanner(System.in);

        System.out.print("Ingresa la cantidad N para la serie: ");
        int n = lec.nextInt();

        for (int i = 1; i <= n; i++) {
            long resultado = (long) Math.pow(i, 2 * i - 1);
            System.out.print("Termino " + i + " de la serie: ");
            System.out.print(resultado + "\n");
        }
        lec.close();
    }
}
